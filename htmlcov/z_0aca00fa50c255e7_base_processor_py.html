42 def rollout(self, x: TensorNTCHW, y: TensorNTCHW | None = None) -> TensorNTCHW: # noqa: ARG002 

43 """Process in latent space across multiple timesteps. 

44 

45 The default implementation simply calls `self.forward` on each time slice until 

46 a sufficient number of forecast steps have been produced. These are then stacked 

47 together to produce the final output. 

48 

49 If you want to handle the NTCHW tensors directly, or to use the target tensor for 

50 model training simply override this method in your child class. 

51 

52 Args: 

53 x: Input TensorNTCHW with (batch_size, n_history_steps, n_latent_channels_total, latent_height, latent_width) 

54 y: during training: Target TensorNTCHW with (batch_size, n_forecast_steps, n_latent_channels_total, latent_height, latent_width) 

55 otherwise: None 

56 

57 Returns: 

58 Predicted TensorNTCHW with (batch_size, n_forecast_steps, n_latent_channels_total, latent_height, latent_width) 

59 

60 """ 

61 # Cut the NTCHW input into NCHW slices 

62 nchw_slices = [x[:, idx_t, :, :, :] for idx_t in range(self.n_history_steps)] 

63 

64 # Rollout the model over the input slices, producing an output for each one. 

65 # Also append the predictions to the list of input slices, so that we can still 

66 # predict when n_forecast_steps > n_history_steps. 

67 outputs: list[TensorNCHW] = [] 

68 for _ in range(self.n_forecast_steps): 

69 outputs.append(self(nchw_slices.pop(0))) 

70 nchw_slices.append(outputs[-1])

73 return stack(outputs, dim=1)
